1.操作系统概述

以分时复用的方式让多个程序可以有效利用计算机的硬件资源，并且相对于程序直接操作硬件资源，通过与操作系统交互间接使用硬件资源的形式更便捷简单。具体地，操作系统以一种称之为 “**_kernel _”的核心程序方式存在。“_kernel _”使用各种 “_interfaces _**”与用户程序交互，详细过程如下图所示：

1.1 进程和内存

这里所讲的进程包括用户空间的内存（instructions，data，stack）和进程状态，“**_kernel _”使用进程标识符 “_PID _”表示进程。进程可以调用接口“_fork() _”来创建子进程，子进程完整复制父进程的内存，值得注意的一点是:二者的内存独立存在，并不共享。在父进程中 “_fork() _”返回子进程的pid，而在子进程中 “_fork() _”返回0，所以我们可以使用这一特点来控制父子进程。进程的结束通过调用 “_exit() _”接口来完成，此时进程会释放所占用的系统资源。父子进程之间的通讯能够利用 “_exit() _”和 “_wait() _”这两个接口配合完成，具体地，在子进程里，将退出状态描述符“0”（成功）或者“1”（失败）利用 “exit(描述符)”传给父进程，父进程通过接口 “_wait() _”接受子进程所传递的状态信号，等待子进程执行完成，随后进行异常处理。在 ”_kernel _“服务中，有一个特殊的进程处理接口 ”_exec() _“。它会直接替换当前进程为指定进程，执行指定进程并且退出，这简化了 “_shell _**”程序的设计。

1.2 I/O和文件描述符

进程的数据传输通过与 “**_kernel _”抽象出来的数据对象”文件描述符“交互实现。文件描述符具体表示含义如下表所示：

文件描述符fd使用 “_open() _”接口创建，进程的数据传输通过 “_read(fd) _”和 “_write(fd) _**”接口来完成。I/O重定向指的是将系统默认的输入输出替换为用户指定的输入输出，这种方法可以实现诸如“ls > a.txt”等操作。I/O重定向的实现离不开fork和exec接口分开调用的原则，这使得文件描述符的替换成为了可能。具体地，按照数据流的不同，在子进程中分别关闭对应的I/O，从而实现I/O的复用。下面给出一个例子：

char *argv[2];
argv[0] = "ls";
argv[1] = 0;
if(fork() == 0){
  close(1);		//释放输出端口
  open("a.txt", O_WRONLY);
  exec("/bin/ls", argv);
  close(1);		//关闭文件
  fprintf(1,"ls wrong\n");	//错误信息显示在屏幕上
  exit(1);
}else{
  wait(0);		//等待子进程正确执行完毕
  fprintf(1,"ls sucess\n");	//输出成功信息
}

“**_dup(fd) _”创建当前文件描述符的一份“引用”，二者共享文件偏移量**。除了fork与dup接口之外，其余任何文件描述符在父子进程中都是独立的。dup的加入，提供了一种显式的端口绑定方法：

// ls > a.txt 2>&1 (2>&1表示标准错误重定向至1输出端口，即将所有信息输出至a.txt）
char *argv[2];
argv[0] = "ls";
argv[1] = 0;
if(fork() == 0){
  close(1);		//释放输出端口
  close(2);		//关闭标准错误
  int fd = open("a.txt", O_WRONLY);
  dup(fd);		//标准错误信息重定向至a.txt
  exec("/bin/ls", argv);
  close(fd);		//关闭文件
  fprintf(1,"ls wrong\n");	//错误信息显示在屏幕上
  exit(1);
}else{
  wait(0);		//等待子进程正确执行完毕
  fprintf(1,"ls sucess\n");	//输出成功信息
}

dup与close的配合使用，也为管道的应用提供了思路。

1.3 管道

管道有一对双向互通的文件描述符构成，我们向管道一端写入数据，从另一端就可以读出传输数据，结合端口重定向和管道，我们能够实现子进程之间数据的传输：

// cat a.txt | wc
int p[2];
char *argv[2],*argv1[2];
argv[0] = "cat a.txt";
argv[1] = 0;
argv1[0] = "wc";
argv1[1] = 0;
pipe(p);
if(fork() == 0){
  fprintf(1,"run cat\n");
  close(1);		//下面4行将标准输出重定向为管道写端，这样执行完exec后，生成的数据会写入管道。
  dup(p[1]);
  close(p[0]);
  close(p[1]);
  exec("/bin/cat", argv);
  fprintf(p[1],"run cat fail!");
  exit(1);
}
if(fork() == 0){
  fprintf(1,"run wc\n");
  close(0);		//将标准输入重定向为管道读端，等待管道数据输入，之后执行exec命令。
  dup(p[0]);
  close(p[0]);
  close(p(1));    
  exec("bin/wc", argv1);
}
close(p[0]);
close(p[1]);
wait(0);
wait(0);

值得注意的是：管道各条子命令分布在独立的子进程中，因此管道命令具有并行的特点。

1.4 文件系统

文件系统包含数据文件，目录，硬件文件三部分，对应的接口有 “**_open() _”, “_mkdir() _”, “_mknod() _**”。一份文件分为文件实体（inode）和文件链接（link），文件实体指的是文件的元信息，例如文件内容、文件的位置、文件长度、文件类型等，文件链接也称为文件名称，一份文件可以有多个名称。对一份文件增加一个名称可以用link()实现，同理，去除一个名称能够用unlink()实现。当文件的链接数为0且无文件描述符引用它时，系统会将其删除，这个机制提供了一种创建临时文件的方法:

fd = open("/tmp/xyz", O_CREATE|O_RDWR);
unlink("/tmp/xyz");

1.5 练习

完成一个乒乓程序，该程序可以在两个进程间实现字节的双向传输。

#include "kernel/types.h"
#include "kernel/stat.h"
#include "user/user.h"

char buf[512];
int msg[512];
int p[2];
int p1[2];


void
ping_pong(int fd, int round)
{
  int n;
  pipe(p);    //parent process write data to child process
  pipe(p1);   //child process write data to parent process

  if(fork() == 0){
    close(p[1]);
    close(p1[0]);
    while((n = read(p[0], buf, sizeof(buf))) > 0){
      sleep(1);
      fprintf(1, "|child  process id: %d, recive data: %s",getpid(), buf); 
      write(p1[1], buf, n);     
      fprintf(1, "|child  process id: %d, send   data: %s", getpid(), buf); 
    }
    exit(0);
  } else {
    close(p[0]);
    close(p1[1]);
    n = read(fd, buf, sizeof(buf));
    fprintf(1, "parent process id: %d, init data: %s, max round num: %d\n", getpid(), buf, round);
    for(int k = 0; k < round; k++){
      fprintf(1, "------------------round: %d------------------\n", k + 1);
      write(p[1], buf, n);
      fprintf(1, "|parent process id: %d, send   data: %s", getpid(), buf);
      sleep(2);
      n = read(p1[0], buf, sizeof(buf));      
      fprintf(1, "|parent process id: %d, recive data: %s", getpid(), buf);  
    }
    fprintf(1, "ping-pong game over\n");
  }
}

int
main(int argc, char *argv[])
{
  int fd, i;
  int round = 4;  //最大回合数

  if(argc <= 1){
    ping_pong(0, round);
    exit(0);
  }
  
  if(argc > 2){
    fprintf(2, "only recive one file!");
    exit(1);
  }
  for(i = 1; i < argc; i++){
    if((fd = open(argv[i], 0)) < 0){
      fprintf(2, "cannot open %s\n", argv[i]);
      exit(1);
    }
    ping_pong(fd, round);
    close(fd);
  }
  exit(0);
}
